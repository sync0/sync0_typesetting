\RedeclareSectionCommands[
% Change the  format of the chapter number.
   tocentrynumberformat=\tocchapternumber, 
% Avoid overfull \hbox message
  tocdynnumwidth,
  tocraggedpagenumber,
  toclinefill=\tocpageseparator,
  tocindent=0em,
  tocpagenumberbox=\tocpagenumberbox
]{chapter}

  tocraggedpagenumber,
  tocbeforeskip=0.1em,
  toclinefill=\tocpageseparator,
